#!/bin/bash

DIR="$( cd "$( dirname "${BASH_SOURCE[0]}" )" && pwd )"
DESKTOP_FILE="$HOME/Desktop/PurWin.desktop"

cat > "$DESKTOP_FILE" <<EOL
[Desktop Entry]
Type=Application
Name=PurWin
Exec=$DIR/launch.sh
Icon=$DIR/icon.png
Path=$DIR
Terminal=false
EOL

chmod +x "$DESKTOP_FILE"
chmod +x "$DIR/launch.sh"

echo "✅ شورتکات ساخته شد و اجرای برنامه از مسیر صحیح تنظیم شد."
